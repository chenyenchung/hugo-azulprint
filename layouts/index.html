"Site" $site "ctx" $ctx -}} {{- if templates.Exists (printf "partials/sections/%s.html" $kind) -}} {{- partial (printf "sections/%s.html" $kind) $entryCtx -}} {{- else -}} {{- errorf "hugo-azulprint: no renderer found for section kind %q — create layouts/partials/sections/%s.html" $kind $kind -}} {{- end -}} {{- end -}} {{ end }}
